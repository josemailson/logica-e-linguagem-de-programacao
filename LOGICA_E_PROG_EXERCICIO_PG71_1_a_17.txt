algoritmo "fatorial

var cont, num, mult: Inteiro

inicio
// Seção de Comandos
  escreva ("Informe um número e conheça o seu fatorial:  ")
  leia (num)

  para cont de num ate 2 passo -1 faca
           se (cont = num) entao
               mult <- cont * (cont - 1)
               escreval ("")
               escreval (cont, " * ", (cont - 1), " = ", mult)
           fimse

           se (cont < num) e (cont <> 1)  entao
                mult <- mult * (cont - 1)
                escreval (mult, " * ", (cont - 1), " = ", mult)
           fimse
   fimpara
           escreval ("")
           escreval ("O fatorial de ", num, " é: ", mult)

 algoritmo "PESQUISA"
var
   nome, sexo, nome_menor: caracter
   idade, qtd_homens, qtd_mulheres, qtd_trab_menor18, qtd_nao_trab, soma_idade: inteiro
   salario, soma_sal_homens, soma_sal_mulheres, menor_renda, pesquisas: real
   trabalha: logico
inicio
   qtd_homens <- 0
   qtd_mulheres <- 0
   qtd_trab_menor18 <- 0
   qtd_nao_trab <- 0
   soma_idade <- 0
   soma_sal_homens <- 0
   soma_sal_mulheres <- 0
   menor_renda <- 999999
   pesquisas <- 0
   escreva("Informe Nome ['FIM' para encerrar]: ")
   leia(nome)
   enquanto (nome <> "FIM") faca
      pesquisas <- pesquisas + 1
      escreva("Informe a idade: ")
      leia(idade)
      enquanto ((idade < 0) ou (idade > 100)) faca
         escreva("Idade inválida. Reinforme a idade: ")
          leia(idade)
      fimenquanto
      escreva("Informe sexo [M/F]: ")
      leia(sexo)
      enquanto ((sexo <> "M") e (sexo <> "F")) faca
         escreva("Valor inválido. Reinforme sexo [M/F]: ")
         leia(sexo)
      fimenquanto
      escreva("Trabalha? [Verdadeiro/Falso]: ")
      leia(trabalha)
      escreva("Informe quantos salários mínimos recebe por mês: ")
      leia(salario)
      se (sexo = "M") entao
         qtd_homens <- qtd_homens + 1
         soma_sal_homens <- soma_sal_homens + salario
      senao
         qtd_mulheres <- qtd_mulheres + 1
         soma_sal_mulheres <- soma_sal_mulheres + salario
      fimse
      se (trabalha) entao
         se (idade < 18) entao
            qtd_trab_menor18 <- qtd_trab_menor18 + 1
         fimse
         se (salario < menor_renda) entao
            menor_renda <- salario
            nome_menor <- nome
         fimse
      senao
         qtd_nao_trab <- qtd_nao_trab + 1
      fimse
      soma_idade <- soma_idade + idade
      escreva("Informe Nome ['FIM' para encerrar]: ")
      leia(nome)
   fimenquanto
   escreval("Quantidade de pesquisas realizadas: ", pesquisas)
   escreval(" - homens: ", qtd_homens)
   escreval(" - mulheres: ", qtd_mulheres)
   escreval("Quantidade pessoas menores de 18 anos que trabalham: ", qtd_trab_menor18)
   escreval("Renda média dos homens: ", soma_sal_homens / qtd_homens)
   escreval("Renda média das mulheres: ", soma_sal_mulheres / qtd_mulheres)
   escreval("Nome e Renda da pessoa que trabalha e que tem a menor renda: ", nome_menor, menor_renda)
   escreval("Quantidade de pessoas que não trabalham: ", qtd_nao_trab)
   escreval("Média de idade dos entrevistados: ", soma_idade / (qtd_homens = qtd_mulheres))
fimalgoritmo
